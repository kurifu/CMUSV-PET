calculate historical data for a given pair of complexity and deliverable type 

     # File app/controllers/deliverable_type_controller.rb, line 142
142:   def calculate_historical_data
143: 
144:     data = []
145:     collector = []
146: 
147:     target_projects = Project.find(:all, :conditions => ['status = ?', 'Archived'])
148:     target_projects.each do |p|
149:       
150:       # Note: do not use <<, as find:all returns an array, we would be appending
151:       # the whole array as a single entry
152:       d = Deliverable.find(:all, :conditions => ['complexity = ? AND deliverable_type = ? AND project_id = ?', session[:complexity], session[:deliverable_type], p.id])
153:       collector = collector | d
154:     end
155: 
156:     sizes = []
157:     efforts = []
158:     rates = []
159:     
160:     unless collector.empty? || collector.blank?
161:       collector.each do |c|
162:         sizes << c.estimated_size
163:         efforts << c.estimated_effort
164:         rates << c.production_rate
165:       end
166: 
167:       data[0] = sizes.min
168:       data[1] = (sizes.size == 0) ? "-" : sizes.sum / sizes.size
169:       data[2] = sizes.max
170: 
171:       data[3] = rates.min
172:       data[4] = (rates.size == 0) ? "-" : rates.sum / rates.size
173:       data[5] = rates.max
174: 
175:       data[6] = efforts.min
176:       data[7] = (efforts.size == 0) ? "-" : efforts.sum / efforts.size
177:       data[8] = efforts.max
178: 
179:     else
180:       flash.now[:warning] = "No data available"
181:       data.fill("?", 0, 9)
182:     end
183: 
184:     return data
185:   end
