View the project overview screen with phase/deliverable estimations 

     # File app/controllers/projects_controller.rb, line 139
139:   def overview
140:     begin
141:       @project = Project.find_by_id(session[:project_id])
142:       @phase_efforts = {}
143: 
144:       phases = RailblazersXmlParser.get_phase(@project.lifecycle)
145:       deliverables = Deliverable.find_all_by_project_id(@project.id)
146: 
147:       # For each phase, create a new Hash if it doesn't exist
148:       0.upto(phases.size-1) do |i|
149:         unless @phase_efforts.has_key?(phases[i])
150:           @phase_efforts[phases[i]] = Hash.new
151: 
152:           # Grab all deliverables in this phase, add it to our small hash
153:           del_to_process = deliverables.select {|d| d.phase == phases[i] }
154:           del_to_process.each do |target|
155:             @phase_efforts[phases[i]][target.name] = target.estimated_effort
156:           end
157:         end
158:       end
159: 
160:       respond_to do |format|
161:         format.html
162:       end
163: 
164:       rescue Exception => ex
165:       @error_msg = ex.message
166: 
167:       render "projects/error"
168:     end
169:   end
